01. [x] 单元格列合并 ​

vue
<template>
  <td
    v-for="(value, key) in item.value"
    :colspan="getColspan(Object.values(getItemsValue(state.items)), item.value)"
  ></td>
</template>

<script>
function getColspan(values, curValue) {
  const curLength = Object.keys(curValue).length;
  const maxLength = values.reduce((max, cur) => {
    return Object.keys(cur).length > max ? Object.keys(cur).length : max;
  }, -Infinity);

  return curLength === 1 && maxLength !== 1 ? maxLength : undefined;
}
</script>
